cmake_minimum_required(VERSION 2.8)

project(libopenssl-download NONE)

find_program(MAKE_EXE NAMES nmake)

SET(CONFIGURE_COMMAND perl
${CMAKE_CURRENT_BINARY_DIR}/build/src/project_libopenssl/Configure no-asm VC-WIN32 --prefix=${OPEN_SRC_INSTALL_PREFIX} --openssldir=${OPEN_SRC_INSTALL_PREFIX})

include(ExternalProject)
ExternalProject_Add(project_libopenssl
    GIT_REPOSITORY    https://github.com/openssl/openssl.git
    GIT_TAG           OpenSSL_1_1_0l
    PREFIX            ${CMAKE_CURRENT_BINARY_DIR}/build
    CONFIGURE_COMMAND ${CONFIGURE_COMMAND}
    BUILD_COMMAND     ${MAKE_EXE}
    BUILD_IN_SOURCE   TRUE
    INSTALL_COMMAND   ${MAKE_EXE} install_sw
    TEST_COMMAND      ""
)
